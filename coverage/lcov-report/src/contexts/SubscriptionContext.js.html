// frontend/src/contexts/SubscriptionContext.js
import React, { createContext, useContext, useState, useEffect, useCallback } from 'react';
import { useAuth } from './AuthContext';
 
// 🔥 HARD-CODED API URL so mobile LAN works reliably
const API_BASE_URL = 'http://192.168.1.185:8000';
 
// Development debug helper (won't trigger ESLint warnings)
const debug = process.env.NODE_ENV === 'development' ? (...args) => console.log(...args) : () => {};
 
const SubscriptionContext = createContext();
export const useSubscription = () => {
  const ctx = useContext(SubscriptionContext);
  if (!ctx) throw new Error('useSubscription must be used within a SubscriptionProvider');
  return ctx;
};
 
export const SubscriptionProvider = ({ children }) => {
  const { token, isAuthenticated } = useAuth();
  const [subscriptionStatus, setSubscriptionStatus] = useState(null);
  const [isLoading, setIsLoading] = useState(false);
  const [tier, setTier] = useState('free');
 
  debug('🔥 SubscriptionContext using API_BASE_URL:', API_BASE_URL);
 
  const getPlanFeatures = useCallback((planTier = tier) => {
    const features = {
      free: {
        name: 'Free', price: 0,
        features: ['5 transcript downloads', 'Basic formats (TXT, SRT, VTT)', 'Community support', 'No priority processing'],
        limits: { clean_transcripts: 5, unclean_transcripts: 3, audio_downloads: 2, video_downloads: 1 }
      },
      pro: {
        name: 'Pro', price: 9.99,
        features: ['100 transcript downloads', 'All formats (TXT, SRT, VTT)', 'Priority processing', 'Email support', 'Advanced transcript cleaning'],
        limits: { clean_transcripts: 100, unclean_transcripts: 50, audio_downloads: 50, video_downloads: 20 }
      },
      premium: {
        name: 'Premium', price: 19.99,
        features: ['Unlimited downloads', 'All formats + API access', 'Fastest processing', 'Priority support', 'Batch processing', 'Custom integrations'],
        limits: { clean_transcripts: Infinity, unclean_transcripts: Infinity, audio_downloads: Infinity, video_downloads: Infinity }
      }
    };
    return features[planTier] || features.free;
  }, [tier]);
 
  const fetchSubscriptionStatus = useCallback(async () => {
    if (!isAuthenticated || !token) {
      setSubscriptionStatus(null); setTier('free'); return;
    }
    try {
      setIsLoading(true);
      const res = await fetch(`${API_BASE_URL}/subscription_status/`, {
        headers: { 'Authorization': `Bearer ${token}` }
      });
      if (res.ok) {
        const data = await res.json();
        data.usage = data.usage || {};
        data.limits = data.limits || getPlanFeatures(data.tier || 'free').limits;
        setSubscriptionStatus(data);
        setTier(data.tier || 'free');
      } else {
        setSubscriptionStatus(null); setTier('free');
      }
    } catch (e) {
      console.error('🔥 Error fetching subscription status:', e);
      setSubscriptionStatus(null); setTier('free');
    } finally {
      setIsLoading(false);
    }
  }, [isAuthenticated, token, getPlanFeatures]);
 
  const refreshSubscriptionStatus = useCallback(async () => { await fetchSubscriptionStatus(); }, [fetchSubscriptionStatus]);
 
  // === Enhanced Checkout helper with better error handling ===
  const startCheckout = useCallback(async (plan /* 'pro' | 'premium' */) => {
    if (!token) throw new Error('Please log in to upgrade your subscription');
    
    const planTarget = (plan || '').toLowerCase();
    if (planTarget !== 'pro' && planTarget !== 'premium') {
      throw new Error('Invalid subscription plan selected');
    }
 
    const lookupKey = planTarget === 'pro' ? 'pro_monthly' : 'premium_monthly';
 
    try {
      const response = await fetch(`${API_BASE_URL}/billing/create_checkout_session`, {
        method: 'POST',
        headers: { 
          'Content-Type': 'application/json', 
          'Authorization': `Bearer ${token}` 
        },
        body: JSON.stringify({ 
          plan: planTarget, 
          price_lookup_key: lookupKey 
        })
      });
 
      // Handle different types of errors
      if (!response.ok) {
        let errorMessage = 'Failed to create checkout session';
        
        try {
          const errorData = await response.json();
          if (errorData?.detail) {
            // Clean up common Stripe error messages for better UX
            if (errorData.detail.includes('No such customer')) {
              errorMessage = 'Account setup issue detected. Please try again - we\'ll fix your account automatically.';
            } else if (errorData.detail.includes('not configured')) {
              errorMessage = 'Payment system is temporarily unavailable. Please try again later.';
            } else {
              errorMessage = errorData.detail;
            }
          }
        } catch (parseError) {
          // If we can't parse the error response, use the status
          if (response.status === 503) {
            errorMessage = 'Payment system is temporarily unavailable. Please try again later.';
          } else if (response.status === 401) {
            errorMessage = 'Please log in again to continue.';
          } else if (response.status >= 500) {
            errorMessage = 'Server error occurred. Please try again in a moment.';
          }
        }
        
        throw new Error(errorMessage);
      }
 
      const data = await response.json();
      
      if (!data?.url) {
        throw new Error('Invalid response from payment system. Please try again.');
      }
 
      // Redirect to Stripe checkout
      window.location.assign(data.url);
      
    } catch (error) {
      // Re-throw the error with context for the UI to handle
      if (error.message) {
        throw error;
      } else {
        throw new Error('Network error occurred. Please check your connection and try again.');
      }
    }
  }, [token]);
 
  // helpers
  const getSafeUsage = useCallback((k) => subscriptionStatus?.usage?.[k] ?? 0, [subscriptionStatus]);
  const getSafeLimit = useCallback((k) => subscriptionStatus?.limits?.[k] ?? getPlanFeatures(tier).limits[k] ?? 0, [subscriptionStatus, tier, getPlanFeatures]);
  const canPerformAction = useCallback((k) => { const u=getSafeUsage(k), l=getSafeLimit(k); return l === 'unlimited' || l === Infinity || u < l; }, [getSafeUsage, getSafeLimit]);
  const hasReachedLimit = useCallback((k)=>!canPerformAction(k),[canPerformAction]);
  const getRemainingUsage = useCallback((k)=>{ const u=getSafeUsage(k), l=getSafeLimit(k); return (l==='unlimited'||l===Infinity)?Infinity:Math.max(0,l-u);},[getSafeUsage,getSafeLimit]);
  const getUsagePercentage = useCallback((k)=>{ const u=getSafeUsage(k), l=getSafeLimit(k); return (l==='unlimited'||l===Infinity)?0:Math.min(100,(u/l)*100);},[getSafeUsage,getSafeLimit]);
  const needsUpgrade = useCallback((k)=>!canPerformAction(k),[canPerformAction]);
  const formatUsage = useCallback((k)=>{ const u=getSafeUsage(k), l=getSafeLimit(k); return (l==='unlimited'||l===Infinity)?`${u} / ∞`:`${u} / ${l}`;},[getSafeUsage,getSafeLimit]);
  const getPlanRecommendation = useCallback(()=>{ if(tier==='premium') return null; const p=['clean_transcripts','unclean_transcripts','audio_downloads','video_downloads'].map(getUsagePercentage); const m=Math.max(...p); return m>80?(tier==='free'?'pro':'premium'):null;},[tier,getUsagePercentage]);
 
  useEffect(() => { 
    if (isAuthenticated && token) {
      fetchSubscriptionStatus(); 
    } else { 
      setSubscriptionStatus(null); 
      setTier('free'); 
    }
  }, [isAuthenticated, token, fetchSubscriptionStatus]);
 
  const value = {
    subscriptionStatus, tier, isLoading,
    refreshSubscriptionStatus,
    startCheckout,
    canPerformAction, hasReachedLimit, getRemainingUsage, getUsagePercentage, needsUpgrade,
    getPlanFeatures, formatUsage, getPlanRecommendation,
    isSubscribed: tier !== 'free', isPro: tier === 'pro', isPremium: tier === 'premium', isFree: tier === 'free',
  };
 
  return <SubscriptionContext.Provider value={value}>{children}</SubscriptionContext.Provider>;
};
 
export default SubscriptionContext;
